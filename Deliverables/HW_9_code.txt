-
dept_no VARCHAR(5) PK

employees
-
emp_no INTEGER PK
emp_title VARCHAR(10) PK
birth_date date
first_name VARCHAR(150)
last_name VARCHAR(200)
sex VARCHAR(5)
hire_date date
last_update timestamp

title VARCHAR(150)
last_update timestamp

salaries
-
id VARCHAR(5) PK

salary INTEGER
last_update timestamp
